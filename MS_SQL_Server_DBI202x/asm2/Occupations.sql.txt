SELECT D.NAME, P.NAME, S.NAME, A.NAME FROM
(SELECT NAME, ROW_NUMBER() OVER(ORDER BY NAME) AS ID FROM OCCUPATIONS WHERE OCCUPATION = 'Doctor') AS D
FULL JOIN
(SELECT NAME, ROW_NUMBER() OVER(ORDER BY NAME) AS ID FROM OCCUPATIONS WHERE OCCUPATION = 'Professor') 
AS P ON D.ID = P.ID
FULL JOIN
(SELECT NAME, ROW_NUMBER() OVER(ORDER BY NAME) AS ID FROM OCCUPATIONS WHERE OCCUPATION = 'Singer') 
AS S ON P.ID = S.ID
FULL JOIN
(SELECT NAME, ROW_NUMBER() OVER(ORDER BY NAME) AS ID FROM OCCUPATIONS WHERE OCCUPATION = 'Actor') 
AS A ON S.ID = A.ID;
