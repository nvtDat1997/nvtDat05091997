SELECT H.HACKER_ID, H.NAME, SUM(TB.SCORE)
FROM HACKERS H, (SELECT HACKER_ID, CHALLENGE_ID, MAX(SCORE) AS SCORE 
FROM SUBMISSIONS GROUP BY HACKER_ID, CHALLENGE_ID) AS TB
WHERE H.HACKER_ID = TB.HACKER_ID
GROUP BY H.HACKER_ID, H.NAME
HAVING SUM(TB.SCORE) > 0
ORDER BY SUM(TB.SCORE) DESC, H.HACKER_ID;
